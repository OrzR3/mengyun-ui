# 快速开始

# 1.安装组件库

npm i mengyun-ui

# 2.引用组件库

在main.js中引用组件库

// 全部引入
import 'mengyun-ui/dist/css/index.css';
import MUI from 'mengyun-ui';
Vue.use(MUI)
// 按需引入
import 'mengyun-ui/dist/css/demo.css';
import { Demo } from 'mengyun-ui';
Vue.use(Demo)
